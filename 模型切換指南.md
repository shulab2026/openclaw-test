  "deepseek-r1:70b:推理"

done

echo ""
read -p "請選擇模型編號 (1-${#MODELS[@]}): " choice

if [[ -n "${MODELS[$((choice-1))]}" ]]; then
  IFS=':' read -r selected_model description <<< "${MODELS[$((choice-1))]}"
  
  echo ""
  echo "⚙️  設定模型：$selected_model ($description)"
  
  # 設定環境變數
  export OLLAMA_MODEL="$selected_model"
  echo "✅ 環境變數已設定：OLLAMA_MODEL=$selected_model"
  
  # 提示永久設定
  read -p "是否永久設定到 ~/.bashrc? (y/n): " confirm
  if [[ "$confirm" == "y" ]]; then
    echo "export OLLAMA_MODEL='$selected_model'" >> ~/.bashrc
    source ~/.bashrc
    echo "✅ 已永久設定到 ~/.bashrc"
  fi
  
  echo ""
  echo "🎉 模型切換完成！請重新啟動你的應用程式。"
else
